Objects
Object Creation:
Object with new constructor
Syntax
let obj = new Object();
Example
let student = new Object();
Console
{ }
Object with new constructor using functions
Syntax
function ObjectName(parameter(s)){
this.parameter = keyName;
};
let newObj =
new ObjectName(parameter(s);
Example
function Student(name, age){
this.name = name
this.age = age;
}

let student1 =
new Student("Sheila", 17);

console.log(student1);
Console
Student {name: 'Sheila', age: 17}
Using Object.create method.
Syntax
let templateObject = {
keyName : valueExample,
keyName : valueExample
};

let objectName =
Object.create(templateObject);
objectName.keyName = value;
Example
let Vegetable = {
name: "",
color: "",
price: 0.00
};

let Vegetable2 =
Object.create(Vegetable);
Vegetable2.name = "pumpkin";
Vegetable2.color = "orange";
Vegetable2.price = 3.43;

console.log(Vegetable2);
Console
{name: 'pumpkin', color: 'orange',
price: 3.43}
color: "orange"
name: "pumpkin"
price: 3.43


Assigning Properties to an Object:
Defining a Property by assigning a value:
Syntax
myObj.keyName = value;
Example
let Cars = new Object();
Cars.color = 'brown';

console.log(Cars);
Console
{color: 'brown'}
Assigning Properties to an Object:
Using an object initializer:
Syntax
let myObj = {
key1: value1,
key2: 'value2'
}
Example
let Cars = {
color: "blue",
year: 2011
};

console.log(Cars);
Console
{color: 'blue'}


Accessing objects:
Use bracket notation, only if key is known at runtime.
let myKey1 = myObj['key1'];
OR
let myKey1 = myObj.key1;
let myCarColor = Cars['color']
OR
let myCarColor = Cars.color;

console.log(myCarColor);
{color: 'pink'}
Assigning values to objects:
myObj['key1'] = 'Value1';
OR
myObj.key1 = 'value1';
Cars['color'] = 'pink';
OR
Cars.color = 'pink';

console.log(Cars);
{color: 'pink'}


Deleting key:values from objects:
delete myObj.key;
OR
delete myObj['key'];
delete Cars.color;
OR
delete Cars['key'];

console.log(Cars);
{year: 1967}
