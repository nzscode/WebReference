Array Methods
let arr1 = [31, 74, 11, 8, 25, 11];
let arr2 = ['p', 'a', 'g', 'u', 'a'];

concat( )

arr1.concat() -->
[31, 74, 11, 8, 25, 11, 'p', 'a', 'g', 'u', 'a']
returns arr1 concatenated to arr2.

join( )

arr1.join("!") --> 31!74!11!8!25!11
returns string by using delimiter between elements of the array.

length( )

arr1.length() --> 5
returns number of indexes in the array.

getindexOf( )

arr2.getindexOf("g") --> 2
returns the index of "g".

lastIndexOf( )

arr1.lastIndexOf("a") --> 4
returns the last index of "a".

sort( )

arr1.sort() --> [11, 11, 25, 31, 74, 8] ~ (11 < 8)
arr2.sort() --> ['a', 'a', 'g', 'p', 'u']
returns array sorted alphabetically.

To sort numerically using function expression:
arr1.sort(function (a, b) {
return a - b;
});
--> [8, 11, 11, 25, 31, 74]

To sort numerically using arrow function:
arr1.sort((a, b) => b - a);
--> [8, 11, 11, 25, 31, 74]

slice( )

arr2.slice(0, 2) --> ['p', 'a']
returns sliced portion of the array including the first index and upto excluding the second index.

pop( )

arr1.pop() --> [31, 74, 11, 8, 25]
returns the string with the last element removed.

push( )

arr1.push(5) --> [31, 74, 11, 8, 25, 11, 5]
returns array with new element appended to end of array.

shift( )

arr1.shift() --> [74, 11, 8, 25, 11]
returns array after removing first element.

unshift( )

arr1.unshift(5) --> [5, 31, 74, 11, 8, 25, 11]
returns array with new element appended to right of array.

toString( )

arr1.toString() --> 31,74,11,8,25,11
returns string after converting array to string.

reverse( )

arr1.reverse() --> [11, 25, 8, 11, 74, 31]
arr2.reverse() --> ['a', 'u', 'g', 'a', 'p']
returns array in reverse.

splice( )

arr1.splice(1, 3) --> [74, 11, 8]
arr2.splice(1, 3) --> ['a', 'g', 'u']
returns array including first index, and total number of index positions to remove.

Array.isArray( )

Array.isArray(arr1) --> true
returns boolean if argument is an array.

valueOf( )

arr1.valueOf()
returns primitive value.
